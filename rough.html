// data.service.ts import { Injectable } from '@angular/core'; import { HttpClient } from '@angular/common/http'; import { Observable } from 'rxjs'; @Injectable({ providedIn: 'root' }) export class DataService { private analysisUrl = 'assets/analysis.json'; private dailyTasksUrl = 'assets/daily-tasks.json'; private shipmentStatsUrl = 'assets/shipment-stats.json'; private blogPostUrl = 'assets/blog-post.json'; constructor(private http: HttpClient) {} getAnalysisData(): Observable { return this.http.get(this.analysisUrl); } getDailyTasksData(): Observable { return this.http.get(this.dailyTasksUrl); } getShipmentStatsData(): Observable { return this.http.get(this.shipmentStatsUrl); } getBlogPostData(): Observable { return this.http.get(this.blogPostUrl); } }
